1. 用 list 存储计算的中间结果

···python
class Solution:
def fib(self, n: int) -> int:
if n == 0:
return 0
elif n == 1:
return 1
else:
tmp = [0,1]
i = 1
while i < n:
tmp.append(tmp[i] + tmp[i-1])
i = i + 1
return tmp.pop() % 1000000007
